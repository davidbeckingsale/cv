\section{Software Projects \texorpdfstring{\hfill{\itshape\footnotesize{}see \url{uk-mac.github.io} to download}}{}}
  \cvitem{CloverLeaf}{A hydrodynamics mini-application to solve the compressible Euler equations in 2D, using an explicit, second-order method.}
  \cvitem{CleverLeaf}{An adaptive mesh refinement implementation of CloverLeaf using the SAMRAI library.}
  \cvitem{TeaLeaf}{A mini-application that solves the linear heat conduction equation on a spatially-decomposed regular grid using a 5 point stencil with implicit solvers.}
  \cvitem{CudaPatchData}{A library of GPU-resident datatypes and adaptive mesh refinement operators for the SAMRAI library.}
